\kafkasubsection{Show that the recursion relations for nonzero magnetic field are
\begin{equation}
    K' \frac{1}{4}\ln \frac{\cosh(2K+h)\cosh(2K-h)}{\cosh^2h}

    h' = h + \frac{1}{2}\ln\ab[\frac{\cosh(2K+h)}{2K-h}]
\end{equation}

\begin{equation}
    \ln A(k,h) = \frac{1}{4}\ln\ab[16\cosh(2K+h)\cosh(2K-h)\cosh^2h]
\end{equation}
}
